create table cart(
	id int(8) NOT NULL primary key AUTO_INCREMENT,
    productid int(3),
    quantity int(3),
    userid int(5),
    foreign key (productid) references products(productid)
);

create table user(
	userid int(5) NOT NULL primary key AUTO_INCREMENT,
    email varchar(50),
    pw varchar(50),
    username varchar(50)
